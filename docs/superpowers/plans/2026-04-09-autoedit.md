# AutoEdit Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an `AutoEdit` command to `lua/ai.lua` that rewrites the current buffer based on a natural-language instruction, shows a diff preview for confirmation, and applies all changes as a single atomic undo unit.

**Architecture:** The AI returns the complete modified file as plain text (full-file-rewrite approach). We compute a line-level LCS diff in Lua, display it in the existing panel.lua right-split with `DiffAdd`/`DiffDelete` highlights, and apply on `y` with a single `nvim_buf_set_lines` call — which Neovim treats as one undo step.

**Tech Stack:** Lua 5.1 (Neovim), Neovim API (`vim.api`, `vim.keymap`), OpenRouter via existing `call_openrouter`, existing `panel.lua` right-split, existing `vim.ui.input` prompt pattern.

- [ ] **Step 2: Insert `myers_diff` into `lua/ai.lua` after `strip_fences`**

```lua
local function myers_diff(a, b)
    local n, m = #a, #b
    local dp = {}
    for i = 0, n do
        dp[i] = {}
        for j = 0, m do
            if i == 0 or j == 0 then
                dp[i][j] = 0
            elseif a[i] == b[j] then
                dp[i][j] = dp[i - 1][j - 1] + 1
            else
                dp[i][j] = math.max(dp[i - 1][j], dp[i][j - 1])
            end
        end
    end
    local ops = {}
    local i, j = n, m
    while i > 0 or j > 0 do
        if i > 0 and j > 0 and a[i] == b[j] then
            table.insert(ops, 1, { op = "keep",   text = a[i] })
            i, j = i - 1, j - 1
        elseif j > 0 and (i == 0 or dp[i][j - 1] >= dp[i - 1][j]) then
            table.insert(ops, 1, { op = "insert", text = b[j] })
            j = j - 1
        else
            table.insert(ops, 1, { op = "delete", text = a[i] })
            i = i - 1
        end
    end
    return ops
end
```

- [ ] **Step 3: Insert `format_diff_lines` immediately after `myers_diff`**

```lua
local function format_diff_lines(ops)
    local CONTEXT = 3
    local display = { "AutoEdit diff \xe2\x80\x94 y to apply, n/q/Esc to cancel", "" }
    -- NOTE: the \xe2\x80\x94 above is the UTF-8 encoding of —. If your editor
    -- converts it, use the literal character: "AutoEdit diff — y to apply, n/q/Esc to cancel"
    local highlights = {}  -- each entry: { 1-based display line index, hl_group }

    local changed = {}
    for idx, op in ipairs(ops) do
        if op.op ~= "keep" then changed[idx] = true end
    end

    local show = {}
    for idx = 1, #ops do
        if changed[idx] then
            for k = math.max(1, idx - CONTEXT), math.min(#ops, idx + CONTEXT) do
                show[k] = true
            end
        end
    end

    local old_n, new_n = 0, 0
    local last_shown = nil

    for idx, op in ipairs(ops) do
        if op.op == "keep" then
            old_n = old_n + 1; new_n = new_n + 1
        elseif op.op == "delete" then
            old_n = old_n + 1
        else
            new_n = new_n + 1
        end

        if show[idx] then
            if last_shown ~= nil and idx > last_shown + 1 then
                table.insert(display, "  ...")
            end
            if op.op == "keep" then
                table.insert(display, string.format("  line %4d   %s", old_n, op.text))
            elseif op.op == "delete" then
                table.insert(display, string.format("- line %4d   %s", old_n, op.text))
                table.insert(highlights, { #display, "DiffDelete" })
            else
                table.insert(display, string.format("+ line %4d   %s", new_n, op.text))
                table.insert(highlights, { #display, "DiffAdd" })
            end
            last_shown = idx
        end
    end

    return display, highlights
end
```

## Task 3: Implement `open_autoedit_diff`

**Files:**
- Modify: `lua/ai.lua` — insert after `format_diff_lines` and before the DISPLAY LAYER comment block

This function: runs the diff, bails if nothing changed, opens the panel, applies highlights, and wires `y`/`n`/`q`/`Esc` keybindings.

- [ ] **Step 1: Insert `open_autoedit_diff` into `lua/ai.lua`**

Insert this function after `format_diff_lines` and before the `-- DISPLAY LAYER` comment:

```lua
local function open_autoedit_diff(old_lines, new_lines, bufnr)
    local ops = myers_diff(old_lines, new_lines)

    local has_changes = false
    for _, op in ipairs(ops) do
        if op.op ~= "keep" then has_changes = true; break end
    end
    if not has_changes then
        vim.notify("AutoEdit: no changes suggested", vim.log.levels.INFO)
        return
    end

    local display, highlights = format_diff_lines(ops)
    require("panel").open(display, { wrap = false })
    local diff_buf = vim.api.nvim_get_current_buf()

    for _, hl in ipairs(highlights) do
        -- hl[1] is 1-based display line index; nvim_buf_add_highlight is 0-based
        vim.api.nvim_buf_add_highlight(diff_buf, -1, hl[2], hl[1] - 1, 0, -1)
    end

    vim.keymap.set("n", "y", function()
        vim.cmd("close")
        vim.schedule(function()
            vim.api.nvim_buf_set_lines(bufnr, 0, -1, false, new_lines)
        end)
    end, { buffer = diff_buf, noremap = true, silent = true })

    vim.keymap.set("n", "n", "<cmd>close<CR>",
        { buffer = diff_buf, noremap = true, silent = true })
end
```

## Task 4: Implement `run_autoedit`, command, and keymaps

**Files:**
- Modify: `lua/ai.lua` — add `run_autoedit` helper and three invocation surfaces in the COMMANDS section

- [ ] **Step 1: Insert `run_autoedit` helper into the COMMANDS section of `lua/ai.lua`**

Insert this before the `vim.api.nvim_create_user_command("AutoEdit", ...)` line you will add in Step 2. Place it after the Autogen command block (after line ~274).

```lua
local function run_autoedit(bufnr, cursor_line, visual_range, task)
    vim.api.nvim_echo(
        { { "  AI: thinking ", "Comment" }, { "▓▓▓▓▓▓▓▓▓▓▓▓", "Comment" } },
        false, {}
    )
    vim.cmd("redraw")

    local old_lines = vim.api.nvim_buf_get_lines(bufnr, 0, -1, false)
    local sys_prompt = load_prompt("autoedit")

    local focus_hint = ""
    if visual_range then
        focus_hint = "\nFocus region is lines " .. visual_range[1] .. "-" .. visual_range[2] .. "."
    end

    local user_msg =
        "--- FILE START ---\n" .. table.concat(old_lines, "\n") .. "\n--- FILE END ---\n\n" ..
        "Cursor is at line " .. cursor_line .. "." .. focus_hint .. "\n\nTask: " .. task

    local result = call_openrouter(sys_prompt, user_msg)
    vim.api.nvim_echo({ { "", "" } }, false, {})  -- clear loading bar

    if not result then return end

    local new_lines = vim.split(result, "\n", { plain = true })
    -- Strip a spurious trailing empty line that some models append
    if new_lines[#new_lines] == "" then table.remove(new_lines) end

    open_autoedit_diff(old_lines, new_lines, bufnr)
end
```

> **Note:** the `▓` loading bar characters are literal UTF-8. If your editor mangles them, copy the string from the existing `<leader>ag` keymap in `lua/ai.lua`.
